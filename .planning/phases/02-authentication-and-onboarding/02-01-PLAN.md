```typescript
import { z } from "zod/v4";

export const loginSchema = z.object({
  email: z.email("Please enter a valid email address."),
  password: z.string().min(1, "Password is required."),
});

export const registerSchema = z.object({
  email: z.email("Please enter a valid email address."),
  password: z.string().min(6, "Password must be at least 6 characters."),
  confirmPassword: z.string(),
}).refine((data) => data.password === data.confirmPassword, {
  message: "Passwords do not match.",
  path: ["confirmPassword"],
});

export const onboardingSchema = z.object({
  plantCountRange: z.enum(["1-5 plants", "6-15 plants", "16-30 plants", "30+ plants"]),
});

export type LoginInput = z.infer<typeof loginSchema>;
export type RegisterInput = z.infer<typeof registerSchema>;
export type OnboardingInput = z.infer<typeof onboardingSchema>;
```

Note: loginSchema password uses `min(1)` (just "required") while registerSchema uses `min(6)` (actual validation). This is intentional — login should not reveal password policy; the auth.ts Credentials provider already validates min(6) server-side.

**4. Update auth.ts — add jwt and session callbacks:**

The existing auth.ts spreads `...authConfig` which includes callbacks.authorized. Add jwt and session callbacks that run ALONGSIDE the authorized callback (not replacing it). The pattern is:

```typescript
import NextAuth from "next-auth";
import Credentials from "next-auth/providers/credentials";
import bcryptjs from "bcryptjs";
import { z } from "zod/v4";
import { authConfig } from "./auth.config";
import { db } from "@/lib/db";

export const { auth, handlers, signIn, signOut } = NextAuth({
  ...authConfig,
  callbacks: {
    ...authConfig.callbacks,
    async jwt({ token, user }) {
      if (user) {
        token.id = user.id;
      }
      return token;
    },
    async session({ session, token }) {
      if (token.id) {
        session.user.id = token.id as string;
      }
      return session;
    },
  },
  providers: [
    Credentials({
      async authorize(credentials) {
        const parsed = z
          .object({
            email: z.email(),
            password: z.string().min(6),
          })
          .safeParse(credentials);

        if (!parsed.success) return null;

        const { email, password } = parsed.data;
        const user = await db.user.findUnique({ where: { email } });
        if (!user) return null;

        const passwordsMatch = await bcryptjs.compare(
          password,
          user.passwordHash,
        );
        if (!passwordsMatch) return null;

        return { id: user.id, email: user.email, name: user.name };
      },
    }),
  ],
});
```

Key change: the `callbacks` object now spreads `...authConfig.callbacks` (preserving `authorized`) and adds `jwt` and `session` callbacks. The Credentials provider remains identical.

**3. Create src/features/auth/actions.ts — Server Actions:**

This file contains two server actions: `registerUser` (AUTH-01, D-03) and `completeOnboarding` (AUTH-04).

CRITICAL: The `signIn` function from NextAuth v5 throws a `NEXT_REDIRECT` error on success. This is NOT a failure — it's how Next.js implements server-side redirects. The try/catch MUST re-throw redirect errors using `isRedirectError()`. Without this, registration appears to succeed (user created in DB) but the browser stays on the register page.

Import path for signIn: `auth.ts` lives at the project root. `actions.ts` is at `src/features/auth/actions.ts`. From that location, three parent traversals reach the project root: `..` = `src/features/`, `../..` = `src/`, `../../..` = project root. Therefore the correct import is `"../../../auth"`.

```typescript
"use server";

import { signIn } from "../../../auth";
import { db } from "@/lib/db";
import bcryptjs from "bcryptjs";
import { registerSchema } from "./schemas";
import { onboardingSchema } from "./schemas";
import { isRedirectError } from "next/dist/client/components/redirect-error";
import { revalidatePath } from "next/cache";

export async function registerUser(data: {
  email: string;
  password: string;
  confirmPassword: string;
}) {
  // 1. Validate input
  const parsed = registerSchema.safeParse(data);
  if (!parsed.success) {
    return { error: "Invalid input." };
  }

  try {
    // 2. Check email uniqueness
    const existingUser = await db.user.findUnique({
      where: { email: parsed.data.email },
    });
    if (existingUser) {
      return { error: "An account with this email already exists. Sign in instead?" };
    }

    // 3. Hash password with bcryptjs (12 rounds)
    const passwordHash = await bcryptjs.hash(parsed.data.password, 12);

    // 4. Create user in database
    await db.user.create({
      data: {
        email: parsed.data.email,
        passwordHash,
      },
    });

    // 5. Auto-login and redirect to dashboard (per D-03)
    // signIn throws NEXT_REDIRECT on success — this is expected behavior
    await signIn("credentials", {
      email: parsed.data.email,
      password: parsed.data.password,
      redirectTo: "/dashboard",
    });
  } catch (error) {
    // CRITICAL: Re-throw redirect errors — they are NOT failures
    if (isRedirectError(error)) {
      throw error;
    }
    return { error: "Something went wrong. Please try again in a moment." };
  }
}

export async function completeOnboarding(data: {
  userId: string;
  plantCountRange: string;
}) {
  // Validate the plant count range
  const parsed = onboardingSchema.safeParse({
    plantCountRange: data.plantCountRange,
  });
  if (!parsed.success) {
    return { error: "Invalid selection." };
  }

  await db.user.update({
    where: { id: data.userId },
    data: {
      onboardingCompleted: true,
      plantCountRange: parsed.data.plantCountRange,
    },
  });

  // Revalidate dashboard so the banner disappears on next server render
  revalidatePath("/dashboard");
  return { success: true };
}
```

Note the error messages match the UI-SPEC Copywriting Contract exactly:
- "An account with this email already exists. Sign in instead?" (register duplicate email)
- "Something went wrong. Please try again in a moment." (network/server error)

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Client form -> Server Action | Untrusted form data crosses from browser to registerUser/completeOnboarding |
| Server Action -> Database | Validated data written to PostgreSQL via Prisma |
| JWT token -> Session | Token claims decoded into session.user for Server Components |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-02-01 | Spoofing | registerUser action | mitigate | Validate all input with Zod registerSchema before any DB operation; reject if safeParse fails |
| T-02-02 | Tampering | completeOnboarding action | mitigate | Validate plantCountRange against z.enum whitelist; userId comes from session (server-side), not client input |
| T-02-03 | Information Disclosure | registerUser error messages | mitigate | Return "An account with this email already exists" for registration (acceptable — user is creating their own account); login uses generic "Incorrect email or password" (never distinguish email-not-found vs wrong-password) |
| T-02-04 | Elevation of Privilege | signIn auto-login after registration | mitigate | signIn("credentials") re-authenticates against the DB (bcryptjs compare) — not a trust escalation since the same password was just used to create the account |
| T-02-05 | Information Disclosure | JWT session.user.id | accept | user.id is a cuid (opaque), not PII. Exposed only in signed httpOnly JWT cookie, not in client-readable state |
| T-02-06 | Denial of Service | registerUser bcryptjs.hash | accept | 12 rounds of bcrypt takes ~250ms per registration. No rate limiting in v1 — documented as known gap. Registration is a low-frequency operation. |
</threat_model>
