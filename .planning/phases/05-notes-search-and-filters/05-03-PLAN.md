```typescript
export async function getPlants(
  userId: string,
  options: {
    roomId?: string;
    search?: string;
    status?: "overdue" | "due-today" | "upcoming" | "archived";
    sort?: "next-watering" | "name" | "recently-added";
    todayStart?: Date;
    todayEnd?: Date;
  } = {}
) {
  const { roomId, search, status, sort, todayStart, todayEnd } = options;

  // Step 1: Archival visibility (per RESEARCH pitfall 1 + pitfall 6)
  // Default (no status param or non-archived status): ALWAYS exclude archived plants
  // Only status=archived shows archived plants
  const archivedFilter =
    status === "archived"
      ? { archivedAt: { not: null } }
      : { archivedAt: null };

  // Step 2: Watering status filter (requires todayStart/todayEnd)
  // Only applied for non-archived statuses that need date comparison
  let statusFilter = {};
  if (status && status !== "archived" && todayStart && todayEnd) {
    if (status === "overdue") {
      statusFilter = { nextWateringAt: { lt: todayStart } };
    } else if (status === "due-today") {
      statusFilter = { nextWateringAt: { gte: todayStart, lt: todayEnd } };
    } else if (status === "upcoming") {
      statusFilter = { nextWateringAt: { gte: todayEnd } };
    }
  }

  // Step 3: Sort mapping (per RESEARCH pitfall 5)
  const orderBy =
    sort === "name" ? { nickname: "asc" as const } :
    sort === "recently-added" ? { createdAt: "desc" as const } :
    { nextWateringAt: "asc" as const }; // default: next-watering

  return db.plant.findMany({
    where: {
      userId,
      ...archivedFilter,
      ...(roomId ? { roomId } : {}),
      ...(search ? {
        OR: [
          { nickname: { contains: search, mode: "insensitive" as const } },
          { species: { contains: search, mode: "insensitive" as const } },
        ],
      } : {}),
      ...statusFilter,
    },
    include: { room: true, careProfile: true },
    orderBy,
  });
}
```

```typescript
"use client";

import { useRouter, useSearchParams } from "next/navigation";
import { useCallback, useRef, useState } from "react";
import { Search, X } from "lucide-react";
import { Input } from "@/components/ui/input";
import { Button } from "@/components/ui/button";
import { Tooltip, TooltipTrigger, TooltipContent } from "@/components/ui/tooltip";

function debounce<T extends (...args: Parameters<T>) => void>(fn: T, delay: number) {
  let timer: ReturnType<typeof setTimeout>;
  return (...args: Parameters<T>) => {
    clearTimeout(timer);
    timer = setTimeout(() => fn(...args), delay);
  };
}

export function SearchBar({ defaultValue }: { defaultValue?: string }) {
  const router = useRouter();
  const searchParams = useSearchParams();
  const [value, setValue] = useState(defaultValue ?? "");
  const inputRef = useRef<HTMLInputElement>(null);

  // eslint-disable-next-line react-hooks/exhaustive-deps
  const updateURL = useCallback(
    debounce((query: string) => {
      const params = new URLSearchParams(searchParams.toString());
      if (query) {
        params.set("search", query);
      } else {
        params.delete("search");
      }
      router.push(`/plants?${params.toString()}`);
    }, 300),
    [searchParams]  // Recreate on searchParams change to avoid stale closure (RESEARCH pitfall 4)
  );

  function handleClear() {
    setValue("");
    const params = new URLSearchParams(searchParams.toString());
    params.delete("search");
    router.push(`/plants?${params.toString()}`);
    inputRef.current?.focus();
  }

  return (
    <div className="relative">
      <Search className="absolute left-sm top-1/2 -translate-y-1/2 h-4 w-4 text-muted-foreground pointer-events-none" />
      <Input
        ref={inputRef}
        placeholder="Search plants..."
        value={value}
        onChange={(e) => {
          setValue(e.target.value);
          updateURL(e.target.value);
        }}
        className="pl-[2.25rem] pr-8 text-sm"
        aria-label="Search plants"
      />
      {value && (
        <Tooltip>
          <TooltipTrigger asChild>
            <Button
              variant="ghost"
              size="icon"
              className="absolute right-sm top-1/2 -translate-y-1/2 h-8 w-8"
              onClick={handleClear}
              aria-label="Clear search"
            >
              <X className="h-4 w-4" />
            </Button>
          </TooltipTrigger>
          <TooltipContent>Clear search</TooltipContent>
        </Tooltip>
      )}
    </div>
  );
}
```

```typescript
import { useRouter, useSearchParams } from "next/navigation";
import { Button } from "@/components/ui/button";

const STATUSES = [
  { value: undefined, label: "All" },
  { value: "overdue", label: "Overdue" },
  { value: "due-today", label: "Due today" },
  { value: "upcoming", label: "Upcoming" },
  { value: "archived", label: "Archived" },
] as const;

export function StatusFilter({ activeStatus }: { activeStatus?: string }) {
  const router = useRouter();
  const searchParams = useSearchParams();

  function handleFilter(status?: string) {
    const params = new URLSearchParams(searchParams.toString());
    if (status) {
      params.set("status", status);
    } else {
      params.delete("status");
    }
    router.push(`/plants?${params.toString()}`);
  }

  return (
    <div className="flex gap-sm overflow-x-auto pb-sm">
      {STATUSES.map((s) => {
        const isActive = s.value === activeStatus || (!s.value && !activeStatus);

        // Color variants per UI-SPEC status filter pill color palette
        let className = "";
        if (isActive && s.value === "overdue") {
          className = "bg-destructive/10 text-destructive border-destructive/20 hover:bg-destructive/15";
        } else if (isActive && s.value === "due-today") {
          className = "bg-accent/15 text-accent border-accent/25 hover:bg-accent/20";
        } else if (!isActive && s.value === "overdue") {
          className = "text-destructive/80 border-destructive/30";
        }

        return (
          <Button
            key={s.value ?? "all"}
            variant={isActive && !className ? "default" : isActive && s.value === "archived" ? "secondary" : "outline"}
            size="sm"
            onClick={() => handleFilter(isActive && s.value ? undefined : s.value)}
            className={className}
          >
            {s.label}
          </Button>
        );
      })}
    </div>
  );
}
```

```tsx
function EmptyFilterState({
  search,
  status,
  roomName,
  hasAnyPlants,
}: {
  search?: string;
  status?: string;
  roomName?: string;
  hasAnyPlants: boolean;
}) {
  // If user has zero plants total (no filters applied), show the original empty state
  // The original empty state is handled by checking plants.length === 0 AND no filters
  // If any filter is active and results are empty, show contextual empty state

  const hasFilters = !!search || !!status || !!roomName;

  if (!hasFilters && !hasAnyPlants) {
    // Original "No plants yet" empty state — already handled in the existing code
    // Return null here; the parent will render the original empty state
    return null;
  }

  // Context-aware empty state per UI-SPEC section 8
  let heading = "No plants found";
  let body = "";
  let ctaLabel = "";
  let ctaParams: Record<string, string | undefined> = {};

  if (status === "archived" && !search && !roomName) {
    heading = "No archived plants";
    body = "Plants you archive will appear here.";
  } else if (search && !status && !roomName) {
    body = `No plants match "${search}". Try a different name or spelling.`;
    ctaLabel = "Clear search";
    ctaParams = { search: undefined };
  } else if (status && !search && !roomName) {
    const statusLabel = status === "due-today" ? "due today" : status;
    body = `No plants are ${statusLabel} right now.`;
    ctaLabel = "Show all plants";
    ctaParams = { status: undefined };
  } else if (roomName && status) {
    const statusLabel = status === "due-today" ? "due today" : status;
    body = `No ${statusLabel} plants in ${roomName}.`;
    ctaLabel = "Clear filters";
    ctaParams = { status: undefined, room: undefined };
  } else {
    body = "No plants match your current filters.";
    ctaLabel = "Clear filters";
    ctaParams = { search: undefined, status: undefined, room: undefined, sort: undefined };
  }

  return (
    <div className="flex flex-col items-center justify-center py-3xl text-center">
      <div className="mb-md rounded-full bg-accent/10 p-lg">
        <SearchIcon className="h-8 w-8 text-accent" />
```

```tsx
// Build clear URL server-side
function buildClearUrl(currentParams: Record<string, string | undefined>, clearKeys: string[]): string {
  const params = new URLSearchParams();
  for (const [key, val] of Object.entries(currentParams)) {
    if (val && !clearKeys.includes(key)) {

  }
```
