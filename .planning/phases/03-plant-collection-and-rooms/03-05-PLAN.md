   ```typescript
   import { auth } from "../../../../../auth";
   import { redirect, notFound } from "next/navigation";
   import { getRoom } from "@/features/rooms/queries";
   import { PlantGrid } from "@/components/plants/plant-grid";
   import { DoorOpen } from "lucide-react";

   export default async function RoomDetailPage({
     params,
   }: {
     params: Promise<{ id: string }>;
   }) {
     const session = await auth();
     if (!session?.user?.id) redirect("/login");

     const { id } = await params;
     const room = await getRoom(id, session.user.id);
     if (!room) notFound();

     return (
       <div className="space-y-lg">
         <div>
           <h1 className="text-xl font-semibold">{room.name}</h1>
           <p className="text-sm text-muted-foreground">
             {room.plants.length} {room.plants.length === 1 ? "plant" : "plants"}
           </p>
         </div>

         {room.plants.length === 0 ? (
           <div className="flex flex-col items-center justify-center py-3xl text-center">

               <DoorOpen className="h-8 w-8 text-accent" />
             </div>
             <h2 className="text-xl font-semibold">No plants in this room</h2>

               Assign plants to this room from their detail page.

         ) : (
   ```

2. Create `src/components/plants/room-filter.tsx` -- a `"use client"` component for the room filter pill bar per D-10:

   ```typescript
   "use client";

   import { useRouter, useSearchParams } from "next/navigation";
   import { Button } from "@/components/ui/button";

   type Room = { id: string; name: string };

   export function RoomFilter({
     rooms,
     activeRoomId,
   }: {
     rooms: Room[];
     activeRoomId?: string;
   }) {
     const router = useRouter();
     const searchParams = useSearchParams();

     function handleFilter(roomId?: string) {
       const params = new URLSearchParams(searchParams.toString());
       if (roomId) {
         params.set("room", roomId);
       } else {
         params.delete("room");
       }
       router.push(`/plants?${params.toString()}`);
     }

     return (
       <div className="flex gap-sm overflow-x-auto pb-sm">
         <Button
           variant={!activeRoomId ? "default" : "outline"}
           size="sm"
           onClick={() => handleFilter()}
         >
           All
         </Button>
         {rooms.map((room) => (
           <Button
             key={room.id}
             variant={activeRoomId === room.id ? "default" : "outline"}
             size="sm"
             onClick={() => handleFilter(room.id)}
           >
             {room.name}
           </Button>
         ))}
       </div>
     );
   }
   ```

   Per UI-SPEC Room Filter Pill Bar spec:
   - Horizontal scrollable `flex gap-sm` above plant grid
   - Each pill: Button with `variant="outline"` default, `variant="default"` when active
   - "All" pill first, then room names in creation order
   - Mobile: `overflow-x-auto` with `pb-sm` for scroll
   - Plain Button pills (NOT shadcn Tabs per UI-SPEC: "Do NOT use shadcn Tabs here")
   - Filter state via URL search param `?room={roomId}` (per research recommendation -- bookmarkable, works with Server Components)
