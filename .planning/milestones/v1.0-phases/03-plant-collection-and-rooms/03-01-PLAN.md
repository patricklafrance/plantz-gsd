3. Create `prisma/seed.ts` -- the seed script that populates CareProfile table using upsert for idempotency:

   ```typescript
   import { PrismaPg } from "@prisma/adapter-pg";
   import { PrismaClient } from "../src/generated/prisma";
   import pg from "pg";
   import { catalogData } from "./data/catalog";

   const pool = new pg.Pool({ connectionString: process.env["DATABASE_URL"] });
   const adapter = new PrismaPg(pool);
   const db = new PrismaClient({ adapter });

   async function main() {
     console.log("Seeding care catalog...");

     for (const entry of catalogData) {
       await db.careProfile.upsert({
         where: { name: entry.name },
         update: {
           species: entry.species,
           wateringInterval: entry.wateringInterval,
           lightRequirement: entry.lightRequirement,
           notes: entry.notes,
         },
         create: {
           name: entry.name,
           species: entry.species,
           wateringInterval: entry.wateringInterval,
           lightRequirement: entry.lightRequirement,
           notes: entry.notes,
         },
       });
     }

     console.log(`Seeded ${catalogData.length} catalog entries.`);
   }

   main()
     .catch((e) => {
       console.error("Seed failed:", e);
       process.exit(1);
     })
     .finally(() => db.$disconnect());
   ```

   Note: Uses relative import `../src/generated/prisma` because the seed script runs from the `prisma/` directory context. Uses `pg` (already installed as dependency of `@prisma/adapter-pg`). The `category` field from catalog data is intentionally NOT stored in the DB -- it's used only for UI grouping.
