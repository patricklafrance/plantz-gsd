  <name>Task 2: Fix focus-after-navigation with MutationObserver</name>
  <files>src/hooks/use-focus-heading.ts</files>
  <action>
Replace the entire contents of `src/hooks/use-focus-heading.ts` to fix UAT Test 8.

The current 50ms setTimeout races against Next.js 16 streaming/Suspense. The h1 heading may not be in the DOM when the timeout fires, causing querySelector to return null and the focus to stay on the clicked nav link.

Replace with a MutationObserver-based approach that waits for the h1 to appear:

```typescript
"use client";
import { useEffect } from "react";
import { usePathname } from "next/navigation";

const HEADING_SELECTOR = "h1[tabindex='-1']";
const MAX_WAIT_MS = 3000;

export function useFocusHeading() {
  const pathname = usePathname();

  useEffect(() => {
    // Try immediately — heading may already be in DOM
    const existing = document.querySelector<HTMLElement>(HEADING_SELECTOR);
    if (existing) {
      existing.focus({ preventScroll: false });
      return;
    }

    // Heading not yet in DOM (streaming/Suspense) — observe for it
    let settled = false;

    const observer = new MutationObserver(() => {
      const h1 = document.querySelector<HTMLElement>(HEADING_SELECTOR);
      if (h1 && !settled) {
        settled = true;
        observer.disconnect();
        h1.focus({ preventScroll: false });
      }
    });

    observer.observe(document.body, {
      childList: true,
      subtree: true,
    });

    // Safety timeout — stop observing after MAX_WAIT_MS
    const timeout = setTimeout(() => {
      if (!settled) {
        settled = true;
        observer.disconnect();
      }
    }, MAX_WAIT_MS);

    return () => {
      settled = true;
      observer.disconnect();
      clearTimeout(timeout);
    };
  }, [pathname]);
}
```

Key design decisions:
- First tries querySelector immediately (handles non-streamed pages with zero delay).
- Falls back to MutationObserver watching `document.body` for childList/subtree changes (catches streamed h1).
- 3-second safety timeout prevents the observer from running indefinitely if no h1 exists on the page.
- Cleanup disconnects observer on route change to prevent stale observers.

**Step 3: Update `src/components/watering/timezone-sync.tsx`**

Replace the entire component. The new logic:
1. Always set the `user_tz` cookie (server pages still need it for date math).
2. Check a `tz_stored` cookie flag. If absent, call the Server Action to persist to DB, then set the flag cookie.
3. This means the DB timezone is only written ONCE (on first visit after the feature deploys, or when the user clears cookies). Subsequent visits only update the cookie.

```typescript
"use client";

import { useEffect } from "react";
import { updateTimezone } from "@/features/auth/actions";

export function TimezoneSync() {
  useEffect(() => {
    const tz = Intl.DateTimeFormat().resolvedOptions().timeZone;

    // Always set cookie for server-side date math
    document.cookie = `user_tz=${encodeURIComponent(tz)}; path=/; SameSite=Strict; max-age=31536000`;

    // Check if we've already persisted TZ to the database in this browser
    const alreadyStored = document.cookie
      .split("; ")
      .some((c) => c.startsWith("tz_stored=1"));

    if (!alreadyStored) {
      // Persist to DB — fire-and-forget
      updateTimezone(tz).then(() => {
        document.cookie = `tz_stored=1; path=/; SameSite=Strict; max-age=31536000`;
      }).catch(() => {
        // Silently fail — will retry on next visit
      });
    }
  }, []);

  return null;
}
```

**Step 4: Update `src/components/shared/timezone-warning.tsx`**

Change TimezoneWarning from a zero-prop client component to one that receives `storedTimezone` as a prop from the server. The comparison is now browser TZ vs DB-stored TZ (not cookie TZ).

```typescript
"use client";

import { useState, useEffect } from "react";
import { AlertTriangle, X } from "lucide-react";
import { Button } from "@/components/ui/button";

interface TimezoneWarningProps {
  storedTimezone: string | null;
}

export function TimezoneWarning({ storedTimezone }: TimezoneWarningProps) {
  const [show, setShow] = useState(false);

  useEffect(() => {
    if (!storedTimezone) return; // No stored TZ yet — skip warning

    const dismissed = sessionStorage.getItem("tz-warning-dismissed");
    if (dismissed) return;

    const browserTz = Intl.DateTimeFormat().resolvedOptions().timeZone;
    if (browserTz !== storedTimezone) {
      setShow(true);
    }
  }, [storedTimezone]);

  if (!show) return null;

  return (
    <div
      role="alert"
      className="flex items-center gap-3 rounded-lg border border-border bg-muted/50 px-4 py-3 text-sm"
    >
      <AlertTriangle className="h-4 w-4 shrink-0 text-muted-foreground" aria-hidden="true" />
      <p className="flex-1 text-muted-foreground">
        Your browser&apos;s timezone differs from our records. Watering dates may display differently.
      </p>
      <Button
        variant="ghost"
        size="sm"
        className="shrink-0 h-8 w-8 p-0"
        onClick={() => {
          setShow(false);
          sessionStorage.setItem("tz-warning-dismissed", "1");
        }}
        aria-label="Dismiss timezone warning"
      >
        <X className="h-4 w-4" />
      </Button>
    </div>
  );
}
```
