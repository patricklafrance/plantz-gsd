**1. Create `src/hooks/use-media-query.ts`** (Client hook):

```tsx
"use client";
import { useState, useEffect } from "react";

export function useMediaQuery(query: string): boolean {
  const [matches, setMatches] = useState(false);

  useEffect(() => {
    const mql = window.matchMedia(query);
    setMatches(mql.matches);
    const handler = (e: MediaQueryListEvent) => setMatches(e.matches);
    mql.addEventListener("change", handler);
    return () => mql.removeEventListener("change", handler);
  }, [query]);

  return matches;
}
```

Note: `useState(false)` means SSR defaults to "not mobile" â€” desktop dialog renders on server, then hydrates to drawer on mobile. This is correct because Server Components don't render modals (they're triggered client-side).

Create responsive wrappers that switch at `(max-width: 639px)`:

```tsx
const MOBILE_QUERY = "(max-width: 639px)";

function ResponsiveDialog(props: { children: React.ReactNode; open?: boolean; onOpenChange?: (open: boolean) => void }) {
  const isMobile = useMediaQuery(MOBILE_QUERY);
  const Root = isMobile ? Drawer : Dialog;
  return <Root {...props} />;
}

function ResponsiveDialogTrigger(props: ...) {
  const isMobile = useMediaQuery(MOBILE_QUERY);
  return isMobile ? <DrawerTrigger {...props} /> : <DialogTrigger {...props} />;
}
```

Create these responsive wrapper components:
- `ResponsiveDialog` (Root)
- `ResponsiveDialogTrigger`
- `ResponsiveDialogContent`
- `ResponsiveDialogHeader`
- `ResponsiveDialogFooter`
- `ResponsiveDialogTitle`
- `ResponsiveDialogDescription`
- `ResponsiveDialogClose`

Each one uses `useMediaQuery(MOBILE_QUERY)` internally and renders the Drawer or Dialog variant accordingly. This pattern allows consumers to do a simple import swap.

**IMPORTANT:** To avoid calling `useMediaQuery` in every sub-component (performance), use a React context:

```tsx
const ResponsiveContext = React.createContext(false);

function ResponsiveDialog({ children, ...props }) {
  const isMobile = useMediaQuery(MOBILE_QUERY);
  const Root = isMobile ? Drawer : Dialog;
  return (
    <ResponsiveContext.Provider value={isMobile}>
      <Root {...props}>{children}</Root>
    </ResponsiveContext.Provider>
  );
}

function ResponsiveDialogContent(props) {
  const isMobile = React.useContext(ResponsiveContext);
  return isMobile ? <DrawerContent {...props} /> : <DialogContent {...props} />;
}
// ... same pattern for all sub-components
```

Export all: `{ ResponsiveDialog, ResponsiveDialogTrigger, ResponsiveDialogContent, ResponsiveDialogHeader, ResponsiveDialogFooter, ResponsiveDialogTitle, ResponsiveDialogDescription, ResponsiveDialogClose }`
