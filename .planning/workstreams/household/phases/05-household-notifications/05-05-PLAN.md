function toCycleEventItem(row, outgoingAssigneeId: string | null): CycleEventItem {
  // priorAssigneeName: find the outgoing assignee's member row in cycle.household.members
  const priorMember = outgoingAssigneeId
    ? row.cycle?.household?.members?.find((m) => m.userId === outgoingAssigneeId)
    : null;
  const priorAssigneeName = priorMember
    ? priorMember.user.name ?? priorMember.user.email
    : null;
  return {
    notificationId: row.id,
    type: row.type as CycleEventItem["type"],
    createdAt: row.createdAt,
    readAt: row.readAt,
    priorAssigneeName,
  };
}
