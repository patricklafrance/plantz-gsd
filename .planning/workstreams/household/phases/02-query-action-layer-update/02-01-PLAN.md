    ```typescript
    import { cache } from "react";
    import { notFound } from "next/navigation";
    import { resolveHouseholdBySlug } from "./queries";
    import { requireHouseholdAccess } from "./guards";

    /**
     * Per-request cached: resolves slug → householdId → membership context.
     * Called from the household layout (mandatory chokepoint, D-03) and from any
     * nested Server Component that needs the household. React.cache() ensures
     * the DB round-trips happen at most once per request.
     *
     * Per D-03/D-18: this is per-request, NOT cross-request — a user removed
     * mid-session sees the change on their next page load. Does NOT cross
     * Server-Component → Server-Action boundary; every mutating action still
     * hits the live DB via requireHouseholdAccess() directly.
     *
     * Throws via notFound() (404) for unknown slugs.
     * Throws ForbiddenError (403) via requireHouseholdAccess for non-members.
     */
    export const getCurrentHousehold = cache(async (slug: string) => {
      const summary = await resolveHouseholdBySlug(slug);
      if (!summary) notFound();
      return await requireHouseholdAccess(summary.id);
    });
    ```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| JWT → session.user.activeHouseholdId | Untrusted client-held token value narrowed to string | undefined at the `session` callback (auth.ts:39) |
| Slug URL → DB | Untrusted URL segment resolved via `resolveHouseholdBySlug` then membership-checked by `requireHouseholdAccess` via `getCurrentHousehold` |
| Migration SQL → DB | Migration file generated by `prisma migrate dev --create-only` (no DB mutation at plan time); backfill SQL is idempotent — `UPDATE ... SET isDefault = true` on a table with exactly one row per user per Phase 1 guarantee |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-02-01-01 | E (Elevation of Privilege) | getCurrentHousehold cache | mitigate | React `cache()` is per-request only — a removed member loses access on the next page load. JSDoc on the function explicitly warns against cross-request or Server-Action reuse. Server Actions in Plan 05 never import `getCurrentHousehold`; they call `requireHouseholdAccess` directly. |
