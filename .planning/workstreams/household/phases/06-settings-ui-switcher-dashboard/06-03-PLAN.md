```typescript
export function buildSwitchPath(currentPathname: string, newSlug: string): string {
  const segments = currentPathname.split("/"); // e.g. ["", "h", "old", "plants", "abc"]
  const detailPattern = /^[a-z0-9]{20,}$/i;
  const isDetailRoute =
    segments.length >= 5 && detailPattern.test(segments[4] ?? "");
  if (isDetailRoute) {
    return `/h/${newSlug}/${segments[3]}`;
  }
  return `/h/${newSlug}/${segments.slice(3).join("/")}`;
}
```

```typescript
const sortedHouseholds = useMemo(
  () =>
    [...households].sort((a, b) => {
      if (a.isDefault !== b.isDefault) return a.isDefault ? -1 : 1;
      // joinedAt isn't in props; stable fallback on slug alphabetical
      return a.household.slug.localeCompare(b.household.slug);
```

CLICK HANDLERS:
```typescript
const [isPending, startTransition] = useTransition();
const pathname = usePathname();
const router = useRouter();

function handleSwitch(newSlug: string) {
  if (newSlug === currentSlug) return;
  router.push(buildSwitchPath(pathname ?? `/h/${currentSlug}/dashboard`, newSlug));
}

function handleSetDefault(householdId: string) {
  startTransition(async () => {
    const result = await setDefaultHousehold({ householdId });
    if ("error" in result) {
      toast.error(result.error);
    } else {
      toast.success("Default household updated.");
    }
  });
}
```
