PAGE SKELETON:
```typescript
type PageProps = {
  params: Promise<{ householdSlug: string }>;
};

export default async function SettingsPage({ params }: PageProps) {
  const { householdSlug } = await params;
  const session = await auth();
  if (!session?.user?.id) redirect("/login");

  const { household, role } = await getCurrentHousehold(householdSlug);

  const [members, availabilities, invitations, counts] = await Promise.all([
    getHouseholdMembers(household.id),
    getHouseholdAvailabilities(household.id),
    role === "OWNER" ? getHouseholdInvitations(household.id) : Promise.resolve([]),
    // Counts for danger zone
    Promise.all([
      db.plant.count({ where: { householdId: household.id } }),
      db.room.count({ where: { householdId: household.id } }),
    ]).then(([plantCount, roomCount]) => ({ plantCount, roomCount })),
  ]);

  const ownerCount = members.filter((m) => m.role === "OWNER").length;
  const memberCount = members.length;

  return (
    <main className="max-w-2xl mx-auto px-4 py-8 space-y-8 pb-20 sm:pb-8">
      <header>
        <h1 className="text-2xl font-semibold" tabIndex={-1}>Settings</h1>
        <p className="text-sm text-muted-foreground mt-1">{household.name}</p>
      </header>

      {/* Section 1: General */}
      <Card>
        <CardHeader className="pb-4">
          <CardTitle className="text-xl font-semibold">General</CardTitle>
        </CardHeader>
        <CardContent className="space-y-4">
          <GeneralForm
            household={{
              id: household.id,
              name: household.name,
              timezone: household.timezone,
              cycleDuration: household.cycleDuration,
            }}
            householdSlug={householdSlug}
            viewerRole={role}
          />
        </CardContent>
      </Card>

      {/* Section 2: Members + Rotation */}
      <Card>
        <CardHeader className="pb-4">
          <CardTitle className="text-xl font-semibold">Members</CardTitle>
          <CardDescription>Rotation order determines who waters this cycle.</CardDescription>
        </CardHeader>
        <CardContent>
          <MembersList
            members={members}
            viewerUserId={session.user.id}
            viewerRole={role}
            householdId={household.id}
            householdSlug={householdSlug}
            householdName={household.name}
            ownerCount={ownerCount}
          />
        </CardContent>
      </Card>

      {/* Section 3: Invitations (OWNER only) */}
      {role === "OWNER" && (
        <Card>
          <CardHeader className="pb-4">
            <CardTitle className="text-xl font-semibold">Invitations</CardTitle>
          </CardHeader>
          <CardContent className="space-y-4">
            <InvitationsCard
              invitations={invitations}
              householdId={household.id}
              householdSlug={householdSlug}
              householdName={household.name}
            />
          </CardContent>
        </Card>
      )}

      {/* Section 4: My Availability */}
      <Card>
        <CardHeader className="pb-4">
          <CardTitle className="text-xl font-semibold">My availability</CardTitle>
          <CardDescription>
            Tell your household when you're unavailable. The rotation will skip your turn during these dates.
          </CardDescription>
        </CardHeader>
        <CardContent className="space-y-4">
          <AvailabilitySection
            availabilities={availabilities}
            viewerUserId={session.user.id}
            viewerRole={role}
            householdId={household.id}
            householdSlug={householdSlug}
          />
        </CardContent>
      </Card>

      {/* Section 5: Danger Zone */}
      <Card className="border-destructive/30">
        <CardHeader>
          <CardTitle className="text-xl font-semibold text-destructive">Danger zone</CardTitle>
        </CardHeader>
        <CardContent className="space-y-4">
          <DangerZoneCard
            viewerRole={role}
            viewerUserId={session.user.id}
            householdId={household.id}
            householdSlug={householdSlug}
            householdName={household.name}
            plantCount={counts.plantCount}
            roomCount={counts.roomCount}
            ownerCount={ownerCount}
            memberCount={memberCount}
          />
        </CardContent>
      </Card>
    </main>
  );
}
```

import { describe, test, expect } from "vitest";
import { readdirSync, readFileSync, statSync } from "node:fs";
import { join } from "node:path";

function walk(dir: string): string[] {
  const files: string[] = [];
  for (const entry of readdirSync(dir)) {
    const fullPath = join(dir, entry);
    if (statSync(fullPath).isDirectory()) {
      files.push(...walk(fullPath));
    } else if (/\.(ts|tsx)$/.test(entry)) {
      files.push(fullPath);
    }
  }
  return files;
}

describe("HSET-01 / Pitfall 17 — internal link prefix audit", () => {
  test("All <Link href=... and router.push targets in phase-6 surfaces use /h/[slug]/ prefix", () => {
    const scanRoots = [
      "src/components/household",
      "src/app/(main)/h",
    ];
    const offenders: Array<{ file: string; match: string; line: number }> = [];

    for (const root of scanRoots) {
      for (const file of walk(root)) {
        const lines = readFileSync(file, "utf8").split("\n");
        lines.forEach((ln, i) => {
          // Match href="/..." or router.push("/...")
          const hrefMatch = ln.match(/href=\{?["`]\/([a-z][a-zA-Z0-9/_\-.]*)/);
          const pushMatch = ln.match(/router\.push\(["`]\/([a-z][a-zA-Z0-9/_\-.]*)/);
          for (const m of [hrefMatch, pushMatch]) {
            if (!m) continue;
            const path = m[1];
            // Allow /h/..., /join/..., /login, /preferences, /api/..., /auth/...
            if (
              path.startsWith("h/") ||
              path.startsWith("join/") ||
              path === "login" ||
              path.startsWith("login") ||
              path.startsWith("preferences") ||
              path.startsWith("api/") ||
              path.startsWith("auth/") ||
              path.startsWith("onboarding")
            ) continue;
            // Template-literal /h/${slug} appears as `/${...` etc. — handled below
            if (ln.includes("/h/${") || ln.includes("/h/$")) continue;
            offenders.push({ file, match: ln.trim(), line: i + 1 });
          }
        });
      }
    }

    if (offenders.length > 0) {
      const report = offenders.map((o) => `  ${o.file}:${o.line} ${o.match}`).join("\n");
      throw new Error(`Pitfall 17 violations (missing /h/[slug]/ prefix):\n${report}`);
    }
    expect(offenders.length).toBe(0);
  });
});
